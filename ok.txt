#!/bin/bash
# Iteration 3: Fully Automated RunPod A40 Setup
# ComfyUI + AnimateDiff + SDXL NSFW + Local Goonsai NSFW LLM
# Workflow JSON auto-import + pre-flight validation

set -e
set -o pipefail

# -------------------------
# Step 1: Update system & install dependencies
# -------------------------
echo "=== Step 1: System update ==="
sudo apt update && sudo apt upgrade -y
sudo apt install -y git wget unzip python3-pip docker.io

# Enable and start Docker
sudo systemctl enable docker
sudo systemctl start docker

# -------------------------
# Step 2: Clone ComfyUI
# -------------------------
echo "=== Step 2: Clone ComfyUI ==="
if [ ! -d "/workspace/ComfyUI" ]; then
    git clone https://github.com/comfyanonymous/ComfyUI.git /workspace/ComfyUI
fi
cd /workspace/ComfyUI

# -------------------------
# Step 3: Install Python dependencies
# -------------------------
echo "=== Step 3: Python setup ==="
python3 -m pip install --upgrade pip
pip install -r requirements.txt

# -------------------------
# Step 4: Install custom nodes
# -------------------------
echo "=== Step 4: Custom nodes ==="
if [ ! -d "custom_nodes/TaraLLM" ]; then
    git clone https://github.com/your-repo/ComfyUI-Tara-LLM-Integration.git custom_nodes/TaraLLM
fi
if [ ! -d "custom_nodes/AnimateDiff" ]; then
    git clone https://github.com/your-repo/ComfyUI-AnimateDiff-Evolved.git custom_nodes/AnimateDiff
fi

# -------------------------
# Step 5: Setup folders & models
# -------------------------
echo "=== Step 5: Folders & models ==="
mkdir -p models/checkpoints models/motion_modules models/vae
mkdir -p outputs/images outputs/videos workflows

# SDXL NSFW model
SDXL_URL="https://huggingface.co/CompVis/stable-diffusion-xl-base-1.0/resolve/main/sd_xl_base_1.0.safetensors"
if [ ! -f "models/checkpoints/sd_xl_base_1.0_nsfw.safetensors" ]; then
    echo "Downloading SDXL NSFW model..."
    wget -O models/checkpoints/sd_xl_base_1.0_nsfw.safetensors $SDXL_URL
fi

# AnimateDiff motion
ANIMATEDIFF_URL="https://huggingface.co/Animation-Mod/mm_sd_v15_v2/resolve/main/mm_sd_v15_v2.ckpt"
if [ ! -f "models/motion_modules/mm_sd_v15_v2.ckpt" ]; then
    echo "Downloading AnimateDiff motion model..."
    wget -O models/motion_modules/mm_sd_v15_v2.ckpt $ANIMATEDIFF_URL
fi

# -------------------------
# Step 6: Install & run Goonsai NSFW LLM locally
# -------------------------
echo "=== Step 6: Goonsai NSFW LLM ==="
if ! docker image inspect goonsai_local:latest >/dev/null 2>&1; then
    echo "Pulling Goonsai NSFW Docker image..."
    docker pull ghcr.io/goonsai/goonsai-nsfw:latest
fi

if ! docker ps --format '{{.Names}}' | grep -q '^goonsai$'; then
    echo "Starting Goonsai NSFW LLM container..."
    docker run -d --name goonsai -p 11434:11434 ghcr.io/goonsai/goonsai-nsfw:latest
fi

# Wait a few seconds for container to start
sleep 5

# Validate LLM endpoint
if curl -s http://localhost:11434/v1 | grep -q "OpenAPI"; then
    echo "Goonsai LLM running correctly."
else
    echo "Warning: Goonsai LLM endpoint not responding."
fi

# -------------------------
# Step 7: Save & embed workflow JSON
# -------------------------
WORKFLOW_JSON="/workspace/ComfyUI/workflows/comfyui_a40_ready.json"
cat > "$WORKFLOW_JSON" << 'EOF'
<PASTE_YOUR_RUNPOD_READY_JSON_HERE>
EOF
echo "Workflow JSON saved to $WORKFLOW_JSON"

# -------------------------
# Step 8: Pre-flight GPU & model check
# -------------------------
echo "=== Step 8: GPU & model validation ==="
if command -v nvidia-smi >/dev/null 2>&1; then
    nvidia-smi
else
    echo "Warning: NVIDIA GPU not detected."
fi

for f in models/checkpoints/sd_xl_base_1.0_nsfw.safetensors models/motion_modules/mm_sd_v15_v2.ckpt; do
    if [ ! -f "$f" ]; then
        echo "Error: Required model missing: $f"
        exit 1
    fi
done
echo "All required models are present."

# -------------------------
# Step 9: Launch ComfyUI with workflow auto-import
# -------------------------
echo "=== Step 9: Launching ComfyUI ==="
# Use --load-workflow to auto-load JSON at startup
python3 main.py --listen 0.0.0.0 --port 8188 --load-workflow "$WORKFLOW_JSON" &

echo "=== Setup Complete ==="
echo "ComfyUI: http://localhost:8188"
echo "LLM API (Goonsai) listening: http://localhost:11434/v1"
echo "Outputs:"
echo "Images -> /workspace/ComfyUI/outputs/images/"
echo "Videos -> /workspace/ComfyUI/outputs/videos/video.webm"
